  6  * @class AI Engine runs the intelligent part of the ai on the server side

  8 AICRAFT.AIEngine = function () {
  9 	this.templateStr = undefined;
 10 	this.ais = new Array();
 11 };
 12 
 13 AICRAFT.AIEngine.prototype = {
 14 
 15 	constructor: AICRAFT.AIEngine,
 16 
 17 	/** load or reload the AI
 18 	 * @param aiStr code string 
 19 	 * @param AIbody physical part 
 20 	 * @param AIname {string} name
 21 	 * Should be called on changed of the code string
 22 	 */
 23 	loadAI: function(aiStr, AIbody, AIname) {
 24 		var templateStr = this.templateStr.replace(/ai_name_to_replace/g, "AI_"+AIname.toString());
 25 		eval(templateStr);
 26 		eval(aiStr);
 27 		eval("var AI = AICRAFT.AI_"+AIname.toString());
 28 		var ai = new AI(AIbody);
 29 		this.ais.push(ai);
 30 	},
 31 
 32 	stepSimulation: function() {
 33 		this.ais.forEach( function(ai) {
 34 			ai.run();
 35 		});
 36 	}
 37 
 38 };
