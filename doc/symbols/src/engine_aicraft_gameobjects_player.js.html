  4 /** @class Player class controls the player models, physic collision bodies and user inputs
  5  * init to be facing north
  6  * @extends AICRAFT.GameObject
  7  * @requires Ammo.js
  8  */
  9 AICRAFT.Player = function (x,y,z,qx,qy,qz,qw, AmmoIn) {
 10 	AICRAFT.GameObject.call(this,x,y,z,qx,qy,qz,qw);
 11 	if (AmmoIn !== undefined) {
 12 		this.Ammo = AmmoIn;
 13 	} else {
 14 		this.Ammo = Ammo;
 15 	}
 16 	this.maxSpeed = 20;
 17 	this.acceleration = 4;
 18 	this.connected = false;
 19 	this.keycode = 0;
 20 	this.codeUploading = false;
 21 };

 47 AICRAFT.Player.prototype.updateInput = function(AmmoIn) {
 48 	if (this.codeUploading === true) {
 49 		return;}
 50 	if (AmmoIn !== undefined) {
 51 		Ammo = AmmoIn;};
 52 	var self = this;
 53 	var velocity = this.phybody.getLinearVelocity();
 54 	var absVelocity = Math.sqrt(velocity.getX()*velocity.getX() + velocity.getY()*velocity.getY() + velocity.getZ()*velocity.getZ()); 
 55 	if (AICRAFT.ClientEngine.key(this.keycode,"w") && absVelocity < this.maxSpeed && this.position.y < 1) {
 56 		AICRAFT.Player.ahead(this, true);}
 57 	if (AICRAFT.ClientEngine.key(this.keycode,"a") && absVelocity < this.maxSpeed && this.position.y < 1) {
 58 		AICRAFT.Player.side(this, true);}
 59 	if (AICRAFT.ClientEngine.key(this.keycode,"s") &&  absVelocity < this.maxSpeed && this.position.y < 1) {
 60 		AICRAFT.Player.ahead(this, false);}
 61 	if (AICRAFT.ClientEngine.key(this.keycode,"d") &&  absVelocity < this.maxSpeed && this.position.y < 1) {
 62 		AICRAFT.Player.side(this, false);}
 63 	if (AICRAFT.ClientEngine.key(this.keycode,"e") && this.position.y < 0.1) {
 64 		this.rotate(2);}
 65 	if (AICRAFT.ClientEngine.key(this.keycode,"q") && this.position.y < 0.1) {
 66 		this.rotate(2, true);}
 67 };

 69 AICRAFT.Player.prototype.rotate = function(degree,IsInverted) {
 70 	if (this.codeUploading === true) {
 71 		return;}
 72 	var self = this;
 73 	if (IsInverted === undefined) {
 74 		IsInverted = false;}
 75 	var ori_quat = self.phybody.getOrientation();
 76 	var quat = new self.Ammo.btQuaternion();
 77 	if (IsInverted === true) {
 78 		quat = AICRAFT.quatFromEuler(degree,0,0);
 79 	} else {
 80 		quat = AICRAFT.quatFromEuler(-1*degree,0,0);
 81 	}
 82 	var result_quat;
 83 	result_quat = AICRAFT.quatMul(ori_quat, quat);
 84 	self.quaternion.x = result_quat.getX();
 85 	self.quaternion.y = result_quat.getY();
 86 	self.quaternion.z = result_quat.getZ();
 87 	self.quaternion.w = result_quat.getW();
 88 	var trans = new self.Ammo.btTransform();	
 89 	trans.setIdentity();
 90 	trans.setOrigin(new self.Ammo.btVector3(self.position.x, self.position.y, self.position.z));
 91 	trans.setRotation(result_quat);
 92 	self.phybody.activate();
 93 	self.phybody.getMotionState().setWorldTransform(trans);
 94 	self.phybody.setCenterOfMassTransform(trans);
 95 };

 97 AICRAFT.Player.side = function(self, IsLeft) {
 98 	var frontVector;
 99 	if (IsLeft === true) {
100 		frontVector = new self.Ammo.btVector3(-1,0,0);
101 	} else {
102 		frontVector = new self.Ammo.btVector3(1,0,0);
103 	}
104 	AICRAFT.Player.move(self, frontVector);
105 };
106 
107 AICRAFT.Player.ahead = function(self, IsAhead) {
108 	var frontVector;
109 	if (IsAhead === true) {
110 		frontVector = new self.Ammo.btVector3(0,0,-1);
111 	} else {
112 		frontVector = new self.Ammo.btVector3(0,0,1);
113 	}
114 	AICRAFT.Player.move(self, frontVector);
115 };
116 
117 AICRAFT.Player.move = function(self, frontVector) {
118 	if (this.codeUploading === true) {
119 		return;}
120 	self.phybody.activate();
121 	var quat = self.phybody.getOrientation();
122 	var transform = new self.Ammo.btTransform();
123 	transform.setIdentity();
124 	transform.setRotation(quat);
125 	frontVector = transform.op_mul(frontVector);
126 	for (var i=0; i<self.acceleration; i++) {
127 		frontVector.setX(frontVector.getX()*1.1);
128 		frontVector.setY(frontVector.getY()*1.1);
129 		frontVector.setZ(frontVector.getZ()*1.1);
130 	};
131 	self.phybody.applyCentralImpulse(frontVector);
132 };
