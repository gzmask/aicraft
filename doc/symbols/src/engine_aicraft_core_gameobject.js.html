  4 /**
  5  * @class GameObjects in the scenes
  6  * @constructor x,y,z are positions and qx, qy, qz, qw are quaternion for rotations
  7  */
  8 AICRAFT.GameObject = function (x,y,z, qx, qy, qz, qw) {
  9 	this.position = new Object();
 10 	this.position.x = parseFloat(x); 
 11 	this.position.y = parseFloat(y); 
 12 	this.position.z = parseFloat(z); 
 13 	this.quaternion = new Object();
 14 	this.quaternion.x = qx || 0;
 15 	this.quaternion.y = qy || 0;
 16 	this.quaternion.z = qz || 0;
 17 	if (qw === undefined) {
 18 		this.quaternion.w = 1 ;
 19 	} else {
 20 		this.quaternion.w = qw;
 21 	}
 22 	this.quaternion.x = parseFloat(this.quaternion.x);
 23 	this.quaternion.y = parseFloat(this.quaternion.y);
 24 	this.quaternion.z = parseFloat(this.quaternion.z);
 25 	this.quaternion.w = parseFloat(this.quaternion.w);
 26 	this.phybody = undefined;
 27 	this.width = 8;
 28 	this.height = 8;
 29 	this.depth = 8;
 30 	this.radius = 5;
 31 	this.mass = 1;
 32 	this.friction = 3;
 33 	this.angularFactor = 0;
 34 	this.IsClient = false;
 35 	this.dynamicsWorld = undefined;
 36 };
 37 
 38 AICRAFT.GameObject.prototype = {
 39 
 40 	constructor: AICRAFT.GameObject,
 41 
 42 	//called by client and server
 43 	buildPhysic: function(AmmoIn, dynamicsWorld) {
 44 		if (AmmoIn !== undefined) {
 45 			Ammo = AmmoIn;
 46 		};
 47 		//var objShape = new Ammo.btBoxShape(new Ammo.btVector3(this.width/2,this.height/2,this.depth/2));
 48 		var objShape = new Ammo.btSphereShape(this.radius);
 49 		//var objShape = new Ammo.btCylinderShape(new Ammo.btVector3(this.radius,this.height/2,this.radius));
 50 		var objTransform = new Ammo.btTransform();	
 51 		objTransform.setIdentity();
 52 		objTransform.setOrigin(new Ammo.btVector3(this.position.x,
 53 			this.position.y,
 54 			this.position.z));
 55 		objTransform.setRotation(new Ammo.btQuaternion(this.quaternion.x, this.quaternion.y, this.quaternion.z, this.quaternion.w));
 56 		var isDynamic = (this.mass != 0);
 57 		var localInertia = new Ammo.btVector3(0,0,0);
 58 		if (isDynamic) {
 59 			objShape.calculateLocalInertia(this.mass,localInertia);}
 60 		var myMotionState = new Ammo.btDefaultMotionState(objTransform);
 61 		var rbInfo = new Ammo.btRigidBodyConstructionInfo(this.mass, myMotionState, objShape, localInertia);
 62 		this.phybody = new Ammo.btRigidBody(rbInfo);
 63 		this.phybody.setFriction(this.friction);
 64 		this.phybody.setAngularFactor(this.angularFactor);
 65 		this.dynamicsWorld = dynamicsWorld;
 66 		this.dynamicsWorld.addRigidBody(this.phybody);
 67 	},
 68 
 69 	//sets the physic states of this object
 70 	setPos: function(AmmoIn,x,y,z,qx,qy,qz,qw,vx,vy,vz) {
 71 		x = parseFloat(x);
 72 		y = parseFloat(y);
 73 		z = parseFloat(z);
 74 		qx = parseFloat(qx);
 75 		qy = parseFloat(qy);
 76 		qz = parseFloat(qz);
 77 		qw = parseFloat(qw);
 78 		vx = parseFloat(vx);
 79 		vy = parseFloat(vy);
 80 		vz = parseFloat(vz);
 81 		if (AmmoIn !== undefined) {
 82 			Ammo = AmmoIn;
 83 		};
 84 		var objTransform = new Ammo.btTransform();	
 85 		objTransform.setIdentity();
 86 		objTransform.setOrigin(new Ammo.btVector3(x, y, z));
 87 		objTransform.setRotation(new Ammo.btQuaternion(qx, qy, qz, qw));
 88 		this.position.x = x;
 89 		this.position.y = y;
 90 		this.position.z = z;
 91 		this.quaternion.x = qx;
 92 		this.quaternion.y = qy;
 93 		this.quaternion.z = qz;
 94 		this.quaternion.w = qw;
 95 		this.phybody.activate();
 96 		this.phybody.getMotionState().setWorldTransform(objTransform);
 97 		this.phybody.setCenterOfMassTransform(objTransform);
 98 		this.phybody.setAngularVelocity(new Ammo.btVector3(vx,vy,vz));
 99 	},
100 
101 	//called by server
102 	physicUpdate: function() {
103 		if (this.phybody.getMotionState()) {
104 			this.phybody.getMotionState().getWorldTransform(this.dynamicsWorld.trans);
105 			this.position.x = parseFloat(this.dynamicsWorld.trans.getOrigin().x().toFixed(2));
106 			this.position.y = parseFloat(this.dynamicsWorld.trans.getOrigin().y().toFixed(2));
107 			this.position.z = parseFloat(this.dynamicsWorld.trans.getOrigin().z().toFixed(2));
108 			this.quaternion.x = parseFloat(this.dynamicsWorld.trans.getRotation().x());
109 			this.quaternion.y = parseFloat(this.dynamicsWorld.trans.getRotation().y());
110 			this.quaternion.z = parseFloat(this.dynamicsWorld.trans.getRotation().z());
111 			this.quaternion.w = parseFloat(this.dynamicsWorld.trans.getRotation().w());
112 		}
113 	}
114 };
