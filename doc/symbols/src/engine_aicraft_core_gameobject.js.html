  4 /**
  5  * @class GameObjects in the scenes
  6  * @constructor x,y,z are positions and qx, qy, qz, qw are quaternion for rotations
  7  */
  8 AICRAFT.GameObject = function (x,y,z, qx, qy, qz, qw) {
  9 	this.position = new Object();
 10 	this.position.x = parseFloat(x); 
 11 	this.position.y = parseFloat(y); 
 12 	this.position.z = parseFloat(z); 
 13 	this.quaternion = new Object();
 14 	this.quaternion.x = qx || 0;
 15 	this.quaternion.y = qy || 0;
 16 	this.quaternion.z = qz || 0;
 17 	if (qw === undefined) {
 18 		this.quaternion.w = 1 ;
 19 	} else {
 20 		this.quaternion.w = qw;
 21 	}
 22 	this.quaternion.x = parseFloat(this.quaternion.x);
 23 	this.quaternion.y = parseFloat(this.quaternion.y);
 24 	this.quaternion.z = parseFloat(this.quaternion.z);
 25 	this.quaternion.w = parseFloat(this.quaternion.w);
 26 	this.mesh = undefined;
 27 	this.phybody = undefined;
 28 	this.width = 8;
 29 	this.height = 8;
 30 	this.depth = 8;
 31 	this.radius = 5;
 32 	this.mass = 1;
 33 	this.friction = 3;
 34 	this.angularFactor = 0;
 35 };
 36 
 37 AICRAFT.GameObject.prototype = {
 38 
 39 	constructor: AICRAFT.GameObject,
 40 
 41 	//called by client
 42 	buildMesh: function(THREE, scene) {
 43 		/*
 44 		this.mesh = new THREE.Mesh(
 45 			new THREE.CubeGeometry(this.width,this.height,this.depth),
 46 			new THREE.MeshLambertMaterial({color: 0xffffff})	
 47 		);
 48 		*/
 49 		/*
 50 		this.mesh = new THREE.Mesh(
 51 			new THREE.CylinderGeometry(this.radius,this.radius,this.height),
 52 			new THREE.MeshLambertMaterial({color: 0xffffff})	
 53 		);
 54 		*/
 55 		this.mesh = new THREE.Mesh(
 56 			new THREE.SphereGeometry(this.radius),
 57 			new THREE.MeshLambertMaterial({color: 0xffffff})	
 58 		);
 59 		this.mesh.castShadow = true;
 60 		this.mesh.receiveShadow = true;
 61 		this.mesh.position.x = this.position.x;
 62 		this.mesh.position.y = this.position.y;
 63 		this.mesh.position.z = this.position.z;
 64 		this.mesh.useQuaternion = true;
 65 		this.mesh.quaternion.set(this.quaternion.x, this.quaternion.y, this.quaternion.z, this.quaternion.w);
 66 		scene.add(this.mesh);
 67 	},	
 68 
 69 	//called by client and server
 70 	buildPhysic: function(AmmoIn) {
 71 		if (AmmoIn !== undefined) {
 72 			Ammo = AmmoIn;
 73 		};
 74 		//var objShape = new Ammo.btBoxShape(new Ammo.btVector3(this.width/2,this.height/2,this.depth/2));
 75 		var objShape = new Ammo.btSphereShape(this.radius);
 76 		//var objShape = new Ammo.btCylinderShape(new Ammo.btVector3(this.radius,this.height/2,this.radius));
 77 		var objTransform = new Ammo.btTransform();	
 78 		objTransform.setIdentity();
 79 		objTransform.setOrigin(new Ammo.btVector3(this.position.x,
 80 			this.position.y,
 81 			this.position.z));
 82 		objTransform.setRotation(new Ammo.btQuaternion(this.quaternion.x, this.quaternion.y, this.quaternion.z, this.quaternion.w));
 83 		var isDynamic = (this.mass != 0);
 84 		var localInertia = new Ammo.btVector3(0,0,0);
 85 		if (isDynamic) {
 86 			objShape.calculateLocalInertia(this.mass,localInertia);}
 87 		var myMotionState = new Ammo.btDefaultMotionState(objTransform);
 88 		var rbInfo = new Ammo.btRigidBodyConstructionInfo(this.mass, myMotionState, objShape, localInertia);
 89 		this.phybody = new Ammo.btRigidBody(rbInfo);
 90 		this.phybody.setFriction(this.friction);
 91 		this.phybody.setAngularFactor(this.angularFactor);
 92 	},
 93 
 94 	//sets the physic states of this object
 95 	setPos: function(AmmoIn,x,y,z,qx,qy,qz,qw,vx,vy,vz) {
 96 		x = parseFloat(x);
 97 		y = parseFloat(y);
 98 		z = parseFloat(z);
 99 		qx = parseFloat(qx);
100 		qy = parseFloat(qy);
101 		qz = parseFloat(qz);
102 		qw = parseFloat(qw);
103 		vx = parseFloat(vx);
104 		vy = parseFloat(vy);
105 		vz = parseFloat(vz);
106 		if (AmmoIn !== undefined) {
107 			Ammo = AmmoIn;
108 		};
109 		var objTransform = new Ammo.btTransform();	
110 		objTransform.setIdentity();
111 		objTransform.setOrigin(new Ammo.btVector3(x, y, z));
112 		objTransform.setRotation(new Ammo.btQuaternion(qx, qy, qz, qw));
113 		this.position.x = x;
114 		this.position.y = y;
115 		this.position.z = z;
116 		this.quaternion.x = qx;
117 		this.quaternion.y = qy;
118 		this.quaternion.z = qz;
119 		this.quaternion.w = qw;
120 		this.phybody.activate();
121 		this.phybody.getMotionState().setWorldTransform(objTransform);
122 		this.phybody.setCenterOfMassTransform(objTransform);
123 		this.phybody.setAngularVelocity(new Ammo.btVector3(vx,vy,vz));
124 	},
125 
126 	//called by client
127 	physicAndGraphicUpdate: function(dynamicsWorld) {
128 		this.physicUpdate.call(this, dynamicsWorld);
129 		this.mesh.position.x = this.position.x;
130 		this.mesh.position.y = this.position.y;
131 		this.mesh.position.z = this.position.z;
132 		this.mesh.quaternion.x = this.quaternion.x;
133 		this.mesh.quaternion.y = this.quaternion.y;
134 		this.mesh.quaternion.z = this.quaternion.z;
135 		this.mesh.quaternion.w = this.quaternion.w;
136 	},
137 
138 	//called by server
139 	physicUpdate: function(dynamicsWorld) {
140 		if (this.phybody.getMotionState()) {
141 			this.phybody.getMotionState().getWorldTransform(dynamicsWorld.trans);
142 			this.position.x = parseFloat(dynamicsWorld.trans.getOrigin().x().toFixed(2));
143 			this.position.y = parseFloat(dynamicsWorld.trans.getOrigin().y().toFixed(2));
144 			this.position.z = parseFloat(dynamicsWorld.trans.getOrigin().z().toFixed(2));
145 			this.quaternion.x = parseFloat(dynamicsWorld.trans.getRotation().x());
146 			this.quaternion.y = parseFloat(dynamicsWorld.trans.getRotation().y());
147 			this.quaternion.z = parseFloat(dynamicsWorld.trans.getRotation().z());
148 			this.quaternion.w = parseFloat(dynamicsWorld.trans.getRotation().w());
149 		}
150 	}
151 };
