 12 AICRAFT.quatMul=function(q1,q2){
 13 	var q3 = new Ammo.btQuaternion(
 14 		q1.getW()*q2.getX() + q1.getX()*q2.getW() + q1.getY()*q2.getZ() - q1.getZ()*q2.getY(),
 15 		q1.getW()*q2.getY() + q1.getY()*q2.getW() + q1.getZ()*q2.getX() - q1.getX()*q2.getZ(),	
 16 		q1.getW()*q2.getZ() + q1.getZ()*q2.getW() + q1.getX()*q2.getY() - q1.getY()*q2.getX(),	
 17 		q1.getW()*q2.getW() - q1.getX()*q2.getX() - q1.getY()*q2.getY() - q1.getZ()*q2.getZ());
 18 	return q3;
 19 };

 21 AICRAFT.quatFromEuler = function (pitch, yaw, roll, AmmoIn){
 22 	if (AmmoIn !== undefined) {
 23 		this.Ammo = AmmoIn;
 24 	} else {
 25 		this.Ammo = Ammo;
 26 	}
 27 	// Basically we create 3 Quaternions, one for pitch, one for yaw, one for roll
 28 	// and multiply those together.
 29 	// the calculation below does the same, just shorter
 30  
 31 	var p = pitch * Math.PI/360;
 32 	var y = yaw * Math.PI/360;
 33 	var r = roll * Math.PI/360;
 34  
 35 	var sinp = Math.sin(p);
 36 	var siny = Math.sin(y);
 37 	var sinr = Math.sin(r);
 38 	var cosp = Math.cos(p);
 39 	var cosy = Math.cos(y);
 40 	var cosr = Math.cos(r);
 41  
 42 	var quat = new this.Ammo.btQuaternion(
 43 		sinr * cosp * cosy - cosr * sinp * siny,
 44 		cosr * sinp * cosy + sinr * cosp * siny,
 45 		cosr * cosp * siny - sinr * sinp * cosy,
 46 		cosr * cosp * cosy + sinr * sinp * siny);
 47  
 48 	return quat.normalize();
 49 };
