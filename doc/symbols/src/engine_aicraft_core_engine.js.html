167 	/** Network initialization
168 	 * Tells the clients all necessary information for game engine initializations
169 	 * @requires socket.io
170 	 * @param socket a opened socket 
171 	 */
172 	networkInit: function(socket) {
173 		var self = this;
174 		//tell client total players
175 		socket.emit('totalPlayers', this.totalPlayers);
176 		//tell client connected players
177 		socket.emit('connect', AICRAFT.Engine.getNextAvailablePnum(this.players));
178 		socket.on('connected', function(data) {
179 			var Pnum = data[0];
180 			self.ais[Pnum].name = data[1];
181 			if (self.players[Pnum].connected || self.players[Pnum] === undefined) {
182 				return false;
183 			};
184 			console.log("Conected players:" + Pnum);
185 			self.players[Pnum].connected = true;
186 			self.aiEngine.initAI(self.ais[Pnum], self.ais[Pnum].name);
187 			socket.set("Pnum", Pnum);
188 		});
189 		//tell client player states
190 		socket.emit('pi', AICRAFT.Engine.encryptedPacket(this.players));
191 		//tell client ai states
192 		socket.emit('ai', AICRAFT.Engine.encryptedPacket(this.ais));
193 		//disconnect peers
194 		socket.on('disconnect', function(){
195 			socket.get('Pnum', function(err,Pnum) {
196 				if (Pnum !== undefined) {
197 					self.players[Pnum].connected = false;}
198 			});
199 		});
200 		//code emitter init
201 		socket.emit('emitterInit', self.aiEngine.templateStr);
202 		//code emitter opened by player
203 		socket.on('code', function(){
204 			socket.get('Pnum', function(err, Pnum) {
205 				self.ais[Pnum].codeUploading = true;
206 				self.players[Pnum].codeUploading = true;
207 			});
208 		});
209 		//code emitter sent in new code by player
210 		socket.on('coded', function(data){
211 			socket.get('Pnum', function(err, Pnum) {
212 				self.ais[Pnum].codeUploading = false;
213 				self.players[Pnum].codeUploading = false;
214 				self.aiEngine.loadAI(data, self.ais[Pnum].name);
215 			});
216 		});
217 	},

283 /**returns a compressed string. 
284  * @requires extractPacket() to extract the info
285  * @param xs is a gameobject array
286  * @return an array contains position, quaternion, velocity
287  */
288 AICRAFT.Engine.encryptedPacket = function(xs){
289 	var result = new Array();
290 	xs.forEach( function(s) {
291 		//first three floatings are position
292 		result.push(parseFloat(s.position.x));
293 		result.push(parseFloat(s.position.y));
294 		result.push(parseFloat(s.position.z));
295 		//next four are quats
296 		result.push(s.quaternion.x);
297 		result.push(s.quaternion.y);
298 		result.push(s.quaternion.z);
299 		result.push(s.quaternion.w);
300 		if (s.sight !== undefined) {
301 			result.push(s.sight.quaternion.x);
302 			result.push(s.sight.quaternion.y);
303 			result.push(s.sight.quaternion.z);
304 			result.push(s.sight.quaternion.w);
305 		}
306 		//last three are velocity
307 		result.push(s.phybody.getAngularVelocity().getX());
308 		result.push(s.phybody.getAngularVelocity().getY());
309 		result.push(s.phybody.getAngularVelocity().getZ());
310 		result.push(s.IsMoving);
311 	});
312 	return result;
313 };

329 				packet[i+2]+'],';
330 			json_text += '"quaternion":';

332 				packet[i+4]+','+
333 				packet[i+5]+','+
334 				packet[i+6]+'],';
335 			json_text += '"velocity":';

337 				packet[i+8]+','+

375 /**
376  * This function returns the next available slot number. If returns -1, game's full
377  * @param players an array of player slots
378  * @return the available index for the player array, or -1 if nothing available
379  */
380 AICRAFT.Engine.getNextAvailablePnum = function(players) {
381 	if (players === undefined) {
382 		return -1;
383 	}
384 	for (var i = 0; i<players.length; i++) {
385 		if (!players[i].connected) {
386 			return i;
387 		}
388 	}
389 	return -1;
390 };
