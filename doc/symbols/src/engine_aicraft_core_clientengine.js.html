241 	networkReady: function(init_cb, animate_cb, syncPos_cb, syncKey_cb) {
242 		var self = this;
243 		self.socket = io.connect('/');
244 		self.socket.on('totalPlayers', function(data) {
245 			self.totalPlayers = data;
246 		});
247 		self.socket.on('connect', function(data) {
248 			self.myPnum = data;
249 		});
250 		//read player status from server
251 		self.socket.on('pi', function(data) {
252 			self.socket.players = AICRAFT.Engine.extractPacket(data);
253 			//read ai status from server
254 			self.socket.on('ai', function(data) {
255 				self.socket.ais = AICRAFT.Engine.extractPacket(data);
256 				if (self.myPnum != -1) {
257 					init_cb(self.socket);
258 					self.players[self.myPnum].connected = true;
259 					var ai_name = "aicraft"+self.myPnum.toString();
260 					do {
261 						prompt("what is the name of your AI?", ai_name);} 
262 					while (self.aiNameExist(ai_name) === true );
263 					self.ais[self.myPnum].name = ai_name;
264 					//finished reading and reported connected
265 					self.socket.emit('connected', [self.myPnum, self.ais[self.myPnum].name]);
266 					animate_cb();
267 					syncPos_cb();
268 					syncKey_cb();
269 				} else {
270 					alert('game is full');
271 				}
272 			});
273 		});
274 	},

316 	syncKey: function() {
317 		var self = this;
318 		AICRAFT.requestKeyFrame(self.syncKey.bind(self), self.keyFPS);
319 		if (self.players[self.myPnum] === undefined || self.myPnum === undefined) {
320 			return;}
321 		if (self.players[self.myPnum].keycode != 0) {
322 			self.socket.emit("k", self.players[self.myPnum].keycode);
323 			self.players[self.myPnum].updateInput(Ammo, self.codeEmitter);
324 		} else if((self.players[self.myPnum].keycode == 0) && (self.lastKeycode != 0)) {
325 			self.socket.emit("k", 0);
326 		};
327 		self.lastKeycode = self.players[self.myPnum].keycode;
328 		/*mouse disabled
329 		if (self.cameraControls.mouseDragOn === true) {
330 			var deltaX = self.cameraControls.deltaX * self.cameraControls.speed;
331 			self.socket.emit("m", deltaX);
332 			self.players[self.myPnum].rotate(deltaX);}
333 			*/
334 	},

336 	animate: function() {
337 		// update delta
338 		this.delta = this.clock.getDelta();
339 
340 		var self = this;
341 
342 		requestAnimationFrame(self.animate.bind(self));
343 
344 		// update physics and graphics
345 		self.dynamicsWorld.stepSimulation(1/self.phyFPS, 10);
346 		(function(){ for (var i=0; i<self.totalPlayers; i++) {
347 			self.players[i].physicAndGraphicUpdate(self.dynamicsWorld, self.delta);
348 			self.ais[i].physicAndGraphicUpdate(self.dynamicsWorld, self.delta);
349 		}})();
350 
351 		// update camera controls
352 		self.cameraControls.update();
353 
354 		// do the render
355 		self.render();
356 
357 		// update stats
358 		self.stats.update();
359 	},

413 /*keyboard input checker
414  * input: keycode and the key you want to know if it's pressed
415  * output: true if it's pressed, false otherwise
416  */
417 AICRAFT.ClientEngine.key = function(keycode, key) {
418 	if (key == "w") {
419 		if (keycode & 8) {
420 			return true;
421 		} else {
422 			return false;
423 		};	
424 	} else if (key == "a") {
425 		if (keycode & 4) {
426 			return true;
427 		} else {
428 			return false;
429 		};	
430 	} else if (key == "s") {
431 		if (keycode & 2) {
432 			return true;
433 		} else {
434 			return false;
435 		};	
436 	} else if (key == "d") {
437 		if (keycode & 1) {
438 			return true;
439 		} else {
440 			return false;
441 		};	
442 	} else if (key == "ctl") {
443 		if (keycode & 64) {
444 			return true;
445 		} else {
446 			return false;
447 		};	
448 	} else if (key == "e") {
449 		if (keycode & 16) {
450 			return true;
451 		} else {
452 			return false;
453 		};	
454 	} else if (key == "q") {
455 		if (keycode & 32) {
456 			return true;
457 		} else {
458 			return false;
459 		};	
460 	};
461 	return false;
462 };
