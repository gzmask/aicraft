  9 AICRAFT.Ai = function (x,y,z,qx,qy,qz,qw, AmmoIn) {
 10 	AICRAFT.GameObject.call(this,x,y,z,qx,qy,qz,qw);
 11 	if (AmmoIn !== undefined) {
 12 		this.Ammo = AmmoIn;
 13 	} else {
 14 		this.Ammo = Ammo;
 15 	}
 16 
 17 	this.sight = new Object();
 18 	this.sight.lines = new Array();
 19 	this.sight.quaternion = new Object();
 20 	this.sight.quaternion.x = 0;
 21 	this.sight.quaternion.y = 0;
 22 	this.sight.quaternion.z = 0;
 23 	this.sight.quaternion.w = 1;
 24 	this.sight.range = 80;
 25 	this.sightMesh = undefined;
 26 	this.maxSpeed = 10;
 27 	this.acceleration = 28;

 29 	this.moveLock = false;

 31 	this.lookAtLock = false;

 44 AICRAFT.Ai.prototype.buildPhysic = function(AmmoIn, dynamicsWorld) {

 47 	this.sight.quaternion.x = this.quaternion.x;

 54 /**
 55  * updates the physical body, the sight, and checks for enemy using raycasting
 56  * 
 57  */
 58 AICRAFT.Ai.prototype.physicUpdate = function() {
 59 	var self = this;
 60 	AICRAFT.GameObject.prototype.physicUpdate.call(self, self.dynamicsWorld);
 61 	//matches the current sight position
 62 	var sight_quat = new self.Ammo.btQuaternion(self.sight.quaternion.x,
 63 			self.sight.quaternion.y,
 64 			self.sight.quaternion.z,
 65 			self.sight.quaternion.w);
 66 	var trans = new self.Ammo.btTransform();
 67 	trans.setIdentity();
 68 	trans.setRotation(sight_quat);
 69 	var targetV3 = new self.Ammo.btVector3(0,0,-1);
 70 	targetV3 = trans.op_mul(targetV3);
 71 	self.sight.lines = AICRAFT.Ai.getSight(self.position.x,self.position.y,self.position.z, 
 72 			targetV3.getX()+self.position.x,targetV3.getY()+self.position.y,targetV3.getZ()+self.position.z, 
 73 			80, 60, 10, self.Ammo, true);
 74 	
 75 
 76 	self.raycast(1000);
 77 	self.Ammo.destroy(trans);
 78 	self.Ammo.destroy(sight_quat);
 79 };

 81 /**
 82  * using raycast to check if enemy is insight. if enemy is found, stop checking for delay value in ms
 83  */
 84 AICRAFT.Ai.prototype.raycast = function(delay) {
 85 	if (this.raycastLock === true) {return;}
 86 	var self = this;
 87 	for (var i=0; i < self.sight.lines.length; i=i+2) {
 88 		var start = self.sight.lines[i];
 89 		var end = self.sight.lines[i+1];
 90 		var cb = new self.Ammo.ClosestRayResultCallback(start, end);
 91 		this.dynamicsWorld.rayTest(start, end, cb);
 92 		if (cb.hasHit()) {
 93 			self.raycastLock = true;
 94 			self.found(cb.get_m_hitPointWorld().getX(),
 95 					cb.get_m_hitPointWorld().getY(),
 96 					cb.get_m_hitPointWorld().getZ(), cb.get_m_collisionObject().getIslandTag());
 97 			setTimeout(function(){
 98 				self.raycastLock = false;
 99 			}, delay);
100 		}
101 	}
102 };

105 /**
106  * fire at target
107  * Ray cast is used to calculate if it's hit.
108  * if hits, apply force to the target, reduce its HP
109  */
110 AICRAFT.Ai.prototype.fireAt = function(x,y,z, fn_cb) {
111 	if (this.weaponLock === true) {return;}
112 	var self = this;
113 	var start = new self.Ammo.btVector3(self.position.x, self.position.y, self.position.z);
114 	var end = new self.Ammo.btVector3(x - self.position.x,
115 			y - self.position.y,
116 			z - self.position.z);
117 	end.normalize();
118 	end.op_mul(self.weaponRange);
119 	end.op_add(start);
120 	console.log('fire from'+start.getX()+','+start.getZ()+' to '+end.getX()+','+end.getZ());
121 	var cb = new self.Ammo.ClosestRayResultCallback(start, end);
122 	self.dynamicsWorld.rayTest(start,end,cb);
123 	if (cb.hasHit()) {
124 		self.weaponLock = true;
125 		//insert bullet hit stuff
126 		console.log('hit! objectID:'+cb.get_m_collisionObject().getIslandTag());
127 		setTimeout(function(){
128 			self.weaponLock = false;
129 			if (fn_cb !== undefined) {fn_cb();}
130 		}, self.weaponDelay);
131 	}
132 };

135  * binds the user found function to this instance
136  */
137 AICRAFT.Ai.prototype.found = function(x,y,z, tag) {
138 	if (tag===-1){return;}
139 	if (tag===this.owner.phybody.getIslandTag()) {return;}
140 	event = new Object();
141 	event.position = [x, y, z];
142 	event.tag = tag;
143 	try {
144 		this.onSightFound(event);
145 	} catch(err) {
146 		return;
147 	}
148 };

180 /** set the sight to a degree related to the front of the AI instantely. Only allow once per period of time
181  * @param to degree that needs to be rotated.
182  * @cb callback function to execute after the lookAt lock is release
183  */
184 AICRAFT.Ai.prototype.lookAt = function(degree, cb) {
185 	var self = this;
186 	if (self.lookAtLock === true) {
187 		return;}	
188 	if (degree > 360) {
189 		degree = degree % 360;}
190 	var front_quat = self.phybody.getOrientation();
191 	var target_quat = AICRAFT.quatFromEuler(degree,0,0,self.Ammo);
192 	var sight_quat = AICRAFT.quatMul(target_quat,front_quat);
193 	self.sight.quaternion.x = sight_quat.getX();
194 	self.sight.quaternion.y = sight_quat.getY();
195 	self.sight.quaternion.z = sight_quat.getZ();
196 	self.sight.quaternion.w = sight_quat.getW();
197 	self.lookAtLock = true;
198 	setTimeout(function(){
199 		self.lookAtLock = false;
200 		if (cb !== undefined) {
201 			cb();}
202 	}, 3000);
203 };

256 /** Rotates the body to a degree related to the front of the AI or
257  * Rotates the sight to a degree related to the front of the Sight
258  * @param degree degrees that needs to be rotated.
259  */
260 AICRAFT.Ai.rotate = function(self, degree, cb, IsLeft, IsBody, IsSight, delay) {
261 	if (degree < 1 || self.hp < 1 || self.codeUploading || self.rotateLock === true) {
262 		if (cb !== undefined && self.codeUploading !== true) {
263 			cb();}
264 		console.log('quiting rotate');
265 		return false;
266 	}
267 	self.rotateLock = true;
268 	if (degree > 360) {
269 		degree = degree % 360;}
270 	var ori_quat = self.phybody.getOrientation();
271 	var sight_quat = new self.Ammo.btQuaternion(self.sight.quaternion.x,
272 		self.sight.quaternion.y,
273 		self.sight.quaternion.z,
274 		self.sight.quaternion.w);
275 	var quat = new self.Ammo.btQuaternion();
276 	if (IsLeft === true) {
277 		quat = AICRAFT.quatFromEuler(1,0,0);
278 	} else {
279 		quat = AICRAFT.quatFromEuler(-1,0,0);
280 	}
281 	var result_quat;
282 	if (IsBody === true) {
283 		result_quat = AICRAFT.quatMul(ori_quat, quat);
284 		self.quaternion.x = result_quat.getX();
285 		self.quaternion.y = result_quat.getY();
286 		self.quaternion.z = result_quat.getZ();
287 		self.quaternion.w = result_quat.getW();
288 		var trans = new self.Ammo.btTransform();	
289 		trans.setIdentity();
290 		trans.setOrigin(new self.Ammo.btVector3(self.position.x, self.position.y, self.position.z));
291 		trans.setRotation(result_quat);
292 		self.phybody.activate();
293 		self.phybody.getMotionState().setWorldTransform(trans);
294 		self.phybody.setCenterOfMassTransform(trans);
295 	} 
296 	if (IsSight === true) {
297 		result_quat = AICRAFT.quatMul(sight_quat, quat);
298 		self.sight.quaternion.x = result_quat.getX();
299 		self.sight.quaternion.y = result_quat.getY();
300 		self.sight.quaternion.z = result_quat.getZ();
301 		self.sight.quaternion.w = result_quat.getW();
302 	} 
303 	setTimeout(function(){
304 		self.rotateLock = false;
305 		AICRAFT.Ai.rotate(self, degree-1, cb, IsLeft, IsBody, IsSight, delay);
306 	}, delay);
307 };

309 AICRAFT.Ai.move = function(self, units, cb, IsAhead, delay) {
310 	if (units < 1 || self.hp < 1 || self.codeUploading || self.moveLock === true) {
311 		if (cb !== undefined && self.codeUploading !== true) {
312 			cb();}
313 		console.log('quiting move');
314 		return false;
315 	}
316 	self.moveLock = true;
317 	var velocity = self.phybody.getLinearVelocity();
318 	var absVelocity = Math.sqrt(velocity.getX()*velocity.getX() + velocity.getY()*velocity.getY() + velocity.getZ()*velocity.getZ()); 
319 
320 	self.phybody.activate();
321 	var quat = self.phybody.getOrientation();
322 
323 	var transform = new self.Ammo.btTransform();
324 	transform.setIdentity();
325 	transform.setRotation(quat);
326 
327 	var frontVector = new self.Ammo.btVector3(0,0,-1);
328 	frontVector = transform.op_mul(frontVector);
329 
330 	for (var i=0; i<self.acceleration; i++) {
331 		frontVector.setX(frontVector.getX()*1.1);
332 		frontVector.setY(frontVector.getY()*1.1);
333 		frontVector.setZ(frontVector.getZ()*1.1);
334 	};
335 
336 	if (!IsAhead) {
337 		frontVector.setX(frontVector.getX()*-1);
338 		frontVector.setY(frontVector.getY()*-1);
339 		frontVector.setZ(frontVector.getZ()*-1);
340 	};
341 
342 	if (absVelocity < self.maxSpeed) {
343 		self.phybody.applyCentralImpulse(frontVector);};
344 
345 	setTimeout( function(){
346 		self.moveLock = false;
347 		AICRAFT.Ai.move(self, units-1, cb, IsAhead, delay);
348 	}, delay);
349 };

351 /** returns the array of sight lines
352  * @param x position x
353  * @param y position y
354  * @param z position z
355  * @param tx target position x
356  * @param ty target position y
357  * @param tz target position z
358  * @param length length of the line
359  * @param fov fov in degrees
360  * @param degreePerLine determines the density of the lines
361  * @param Ammo
362  * @param IsAmmo If the return array are Ammo vertices or ThreeJS vertices
363  */
364 AICRAFT.Ai.getSight = function(x,y,z, tx,ty,tz, length, fov, degreePerLine, Ammo, IsAmmo) {
365 	//console.log('==========');
366 	var resultAr = new Array();
367 	//make local coords
368 	tx -= x;
369 	ty -= y;
370 	tz -= z;
371 	
372 	//calculate first target vector from the left
373 	var frontV = new Ammo.btVector3(tx,ty,tz);
374 	var quatRot = AICRAFT.quatFromEuler(fov/2, 0, 0, Ammo);
375 	var transRot = new Ammo.btTransform();
376 	transRot.setRotation(quatRot);
377 	frontV = transRot.op_mul(frontV);
378 	frontV.normalize();
379 	frontV.op_mul(length);
380 	
381 	do {
382 		//console.log('x:'+frontV.getX(), 'y:'+frontV.getY(), 'z:'+frontV.getZ());
383 		resultAr.push(
384 				AICRAFT.v( x,y,z, IsAmmo),
385 				AICRAFT.v( frontV.getX()+x, frontV.getY()+y, frontV.getZ()+z, IsAmmo));
386 		quatRot = AICRAFT.quatFromEuler(-1*degreePerLine, 0, 0, Ammo);
387 		transRot.setRotation(quatRot);
388 		frontV = transRot.op_mul(frontV);
389 		frontV.normalize();
390 		frontV.op_mul(length);
391 		fov -= degreePerLine;
392 	} while (fov >= 0);
393 	
394 	Ammo.destroy(quatRot);
395 	Ammo.destroy(transRot);
396 	return resultAr;
397 };
