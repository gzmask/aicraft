121 /**
122  * updates the physical body, the sight, and checks for enemy using raycasting
123  * 
124  */
125 AICRAFT.Ai.prototype.physicUpdate = function() {
126 	var self = this;
127 	AICRAFT.GameObject.prototype.physicUpdate.call(self, self.dynamicsWorld);
128 	//matches the current sight position
129 	var sight_quat = new self.Ammo.btQuaternion(self.sight.quaternion.x,
130 			self.sight.quaternion.y,
131 			self.sight.quaternion.z,
132 			self.sight.quaternion.w);
133 	var trans = new self.Ammo.btTransform();
134 	trans.setIdentity();
135 	trans.setRotation(sight_quat);
136 	var targetV3 = new self.Ammo.btVector3(0,0,-1);
137 	targetV3 = trans.op_mul(targetV3);
138 	self.sight.lines = AICRAFT.Ai.getSight(self.position.x,self.position.y,self.position.z, 
139 			targetV3.getX()+self.position.x,targetV3.getY()+self.position.y,targetV3.getZ()+self.position.z, 
140 			80, 60, 10, self.Ammo, true);
141 	
142 
143 	self.raycast(1000);
144 	self.Ammo.destroy(trans);
145 	self.Ammo.destroy(sight_quat);
146 };

148 /**
149  * using raycast to check if enemy is insight. if enemy is found, stop checking for delay value in ms
150  */
151 AICRAFT.Ai.prototype.raycast = function(delay) {
152 	if (this.raycastLock === true) {return;}
153 	var self = this;
154 	for (var i=0; i < self.sight.lines.length; i=i+2) {
155 		var start = self.sight.lines[i];
156 		var end = self.sight.lines[i+1];
157 		var cb = new self.Ammo.ClosestRayResultCallback(start, end);
158 		this.dynamicsWorld.rayTest(start, end, cb);
159 		if (cb.hasHit()) {
160 			self.raycastLock = true;
161 			self.found(cb.get_m_hitPointWorld().getX(),
162 					cb.get_m_hitPointWorld().getY(),
163 					cb.get_m_hitPointWorld().getZ(), cb.get_m_collisionObject().getIslandTag());
164 			setTimeout(function(){
165 				self.raycastLock = false;
166 			}, delay);
167 		}
168 	}
169 };

172 /**
173  * fire at target
174  * Ray cast is used to calculate if it's hit.
175  * if hits, apply force to the target, reduce its HP
176  */
177 AICRAFT.Ai.prototype.fireAt = function(x,y,z, fn_cb) {
178 	if (this.weaponLock === true) {return;}
179 	var self = this;
180 	var start = new self.Ammo.btVector3(self.position.x, self.position.y, self.position.z);
181 	var end = new self.Ammo.btVector3(x - self.position.x,
182 			y - self.position.y,
183 			z - self.position.z);
184 	end.normalize();
185 	end.op_mul(self.weaponRange);
186 	end.op_add(start);
187 	console.log('fire from'+start.getX()+','+start.getZ()+' to '+end.getX()+','+end.getZ());
188 	var cb = new self.Ammo.ClosestRayResultCallback(start, end);
189 	self.dynamicsWorld.rayTest(start,end,cb);
190 	if (cb.hasHit()) {
191 		self.weaponLock = true;
192 		//insert bullet hit stuff
193 		console.log('hit! objectID:'+cb.get_m_collisionObject().getIslandTag());
194 		setTimeout(function(){
195 			self.weaponLock = false;
196 			if (fn_cb !== undefined) {fn_cb();}
197 		}, self.weaponDelay);
198 	}
199 };

205 	if (tag===-1){return;}

207 	event = new Object();
208 	event.position = [x, y, z];
209 	event.tag = tag;
210 	try {
211 		this.onSightFound(event);
212 	} catch(err) {
213 		return;
214 	}

247 /** set the sight to a degree related to the front of the AI instantely. Only allow once per period of time
248  * @param to degree that needs to be rotated.
249  * @cb callback function to execute after the lookAt lock is release
250  */
251 AICRAFT.Ai.prototype.lookAt = function(degree, cb) {
252 	var self = this;
253 	if (self.lookAtLock === true) {
254 		return;}	
255 	if (degree > 360) {
256 		degree = degree % 360;}
257 	var front_quat = self.phybody.getOrientation();
258 	var target_quat = AICRAFT.quatFromEuler(degree,0,0,self.Ammo);
259 	var sight_quat = AICRAFT.quatMul(target_quat,front_quat);
260 	self.sight.quaternion.x = sight_quat.getX();
261 	self.sight.quaternion.y = sight_quat.getY();
262 	self.sight.quaternion.z = sight_quat.getZ();
263 	self.sight.quaternion.w = sight_quat.getW();
264 	self.lookAtLock = true;
265 	setTimeout(function(){
266 		self.lookAtLock = false;
267 		if (cb !== undefined) {
268 			cb();}
269 	}, 3000);
270 };

323 /** Rotates the body to a degree related to the front of the AI or
324  * Rotates the sight to a degree related to the front of the Sight
325  * @param degree degrees that needs to be rotated.
326  */
327 AICRAFT.Ai.rotate = function(self, degree, cb, IsLeft, IsBody, IsSight, delay) {
328 	if (degree < 1 || self.hp < 1 || self.codeUploading || self.rotateLock === true) {
329 		if (cb !== undefined && self.codeUploading !== true) {
330 			cb();}
331 		console.log('quiting rotate');
332 		return false;
333 	}
334 	self.rotateLock = true;
335 	if (degree > 360) {
336 		degree = degree % 360;}
337 	var ori_quat = self.phybody.getOrientation();
338 	var sight_quat = new self.Ammo.btQuaternion(self.sight.quaternion.x,
339 		self.sight.quaternion.y,
340 		self.sight.quaternion.z,
341 		self.sight.quaternion.w);
342 	var quat = new self.Ammo.btQuaternion();
343 	if (IsLeft === true) {
344 		quat = AICRAFT.quatFromEuler(1,0,0);
345 	} else {
346 		quat = AICRAFT.quatFromEuler(-1,0,0);
347 	}
348 	var result_quat;
349 	if (IsBody === true) {
350 		result_quat = AICRAFT.quatMul(ori_quat, quat);
351 		self.quaternion.x = result_quat.getX();
352 		self.quaternion.y = result_quat.getY();
353 		self.quaternion.z = result_quat.getZ();
354 		self.quaternion.w = result_quat.getW();
355 		var trans = new self.Ammo.btTransform();	
356 		trans.setIdentity();
357 		trans.setOrigin(new self.Ammo.btVector3(self.position.x, self.position.y, self.position.z));
358 		trans.setRotation(result_quat);
359 		self.phybody.activate();
360 		self.phybody.getMotionState().setWorldTransform(trans);
361 		self.phybody.setCenterOfMassTransform(trans);
362 	} 
363 	if (IsSight === true) {
364 		result_quat = AICRAFT.quatMul(sight_quat, quat);
365 		self.sight.quaternion.x = result_quat.getX();
366 		self.sight.quaternion.y = result_quat.getY();
367 		self.sight.quaternion.z = result_quat.getZ();
368 		self.sight.quaternion.w = result_quat.getW();
369 	} 
370 	setTimeout(function(){
371 		self.rotateLock = false;
372 		AICRAFT.Ai.rotate(self, degree-1, cb, IsLeft, IsBody, IsSight, delay);
373 	}, delay);
374 };

376 AICRAFT.Ai.move = function(self, units, cb, IsAhead, delay) {
377 	if (units < 1 || self.hp < 1 || self.codeUploading || self.moveLock === true) {
378 		if (cb !== undefined && self.codeUploading !== true) {
379 			cb();}
380 		console.log('quiting move');
381 		return false;
382 	}
383 	self.moveLock = true;
384 	var velocity = self.phybody.getLinearVelocity();
385 	var absVelocity = Math.sqrt(velocity.getX()*velocity.getX() + velocity.getY()*velocity.getY() + velocity.getZ()*velocity.getZ()); 
386 
387 	self.phybody.activate();
388 	var quat = self.phybody.getOrientation();
389 
390 	var transform = new self.Ammo.btTransform();
391 	transform.setIdentity();
392 	transform.setRotation(quat);
393 
394 	var frontVector = new self.Ammo.btVector3(0,0,-1);
395 	frontVector = transform.op_mul(frontVector);
396 
397 	for (var i=0; i<self.acceleration; i++) {
398 		frontVector.setX(frontVector.getX()*1.1);
399 		frontVector.setY(frontVector.getY()*1.1);
400 		frontVector.setZ(frontVector.getZ()*1.1);
401 	};
402 
403 	if (!IsAhead) {
404 		frontVector.setX(frontVector.getX()*-1);
405 		frontVector.setY(frontVector.getY()*-1);
406 		frontVector.setZ(frontVector.getZ()*-1);
407 	};
408 
409 	if (absVelocity < self.maxSpeed) {
410 		self.phybody.applyCentralImpulse(frontVector);};
411 
412 	setTimeout( function(){
413 		self.moveLock = false;
414 		AICRAFT.Ai.move(self, units-1, cb, IsAhead, delay);
415 	}, delay);
416 };

418 //animation loader
419 AICRAFT.Ai.JSONloader = function(self, url, IsWalk, scene, color, THREE, cb) {
420 	var loader = new THREE.JSONLoader();
421 	loader.load( url, function(geometry){
422 		var mesh;
423 		var material = geometry.materials[ 0 ];
424 		material.morphTargets = true;
425 		material.color.setHex( color );
426 		material.ambient.setHex( 0x222222 );
427 		var faceMaterial = new THREE.MeshFaceMaterial();
428 		var morph = new THREE.MorphAnimMesh( geometry, faceMaterial );
429 		morph.duration = 1000;
430 		morph.time = 0; //stands for when I am at in the duration
431 		mesh = morph;
432 		mesh.castShadow = true;
433 		mesh.receiveShadow = true;
434 		mesh.position.x = self.position.x;
435 		mesh.position.y = self.position.y;
436 		mesh.position.z = self.position.z;
437 		mesh.useQuaternion = true;
438 		mesh.quaternion.set(self.quaternion.x, self.quaternion.y, self.quaternion.z, self.quaternion.w);
439 		mesh.scale.set(5, 5, 5);
440 		if (IsWalk === true)
441 			self.walkMesh = mesh;
442 		else
443 			self.turnMesh = mesh;
444 		self.mesh = mesh;
445 		scene.add( mesh );
446 		mesh.visible = false;
447 		if (cb !== undefined) {cb();}
448 	}); 
449 };

451 /** returns the array of sight lines
452  * @param x position x
453  * @param y position y
454  * @param z position z
455  * @param tx target position x
456  * @param ty target position y
457  * @param tz target position z
458  * @param length length of the line
459  * @param fov fov in degrees
460  * @param degreePerLine determines the density of the lines
461  * @param Ammo
462  * @param IsAmmo If the return array are Ammo vertices or ThreeJS vertices
463  */
464 AICRAFT.Ai.getSight = function(x,y,z, tx,ty,tz, length, fov, degreePerLine, Ammo, IsAmmo) {
465 	//console.log('==========');
466 	var resultAr = new Array();
467 	//make local coords
468 	tx -= x;
469 	ty -= y;
470 	tz -= z;
471 	
472 	//calculate first target vector from the left
473 	var frontV = new Ammo.btVector3(tx,ty,tz);
474 	var quatRot = AICRAFT.quatFromEuler(fov/2, 0, 0, Ammo);
475 	var transRot = new Ammo.btTransform();
476 	transRot.setRotation(quatRot);
477 	frontV = transRot.op_mul(frontV);
478 	frontV.normalize();
479 	frontV.op_mul(length);
480 	
481 	do {
482 		//console.log('x:'+frontV.getX(), 'y:'+frontV.getY(), 'z:'+frontV.getZ());
483 		resultAr.push(
484 				AICRAFT.v( x,y,z, IsAmmo),
485 				AICRAFT.v( frontV.getX()+x, frontV.getY()+y, frontV.getZ()+z, IsAmmo));
486 		quatRot = AICRAFT.quatFromEuler(-1*degreePerLine, 0, 0, Ammo);
487 		transRot.setRotation(quatRot);
488 		frontV = transRot.op_mul(frontV);
489 		frontV.normalize();
490 		frontV.op_mul(length);
491 		fov -= degreePerLine;
492 	} while (fov >= 0);
493 	
494 	Ammo.destroy(quatRot);
495 	Ammo.destroy(transRot);
496 	return resultAr;
497 };
