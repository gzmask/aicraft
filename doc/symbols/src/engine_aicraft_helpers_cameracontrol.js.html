1 AICRAFT.CameraControl = function(camera, gameObj, domElement) {
  2 	this.camera	= camera;
  3 	this.camera.useQuaternion = true;
  4 	this.camera.lookAt(new THREE.Vector3(0,0,-1));
  5 	this.domElement	= domElement || document;
  6 	this.gameObj	= gameObj;
  7 	this.target	= new THREE.Vector3(0, 0, 0);
  8 	this.mouseX = 0;
  9 	this.prevMouseX = 0;
 10 	this.deltaX = 0;
 11 	this.mouseY = 0;
 12 	this.prevMouseY = 0;
 13 	this.deltaY = 0;
 14 	this.mouseDragOn = false;
 15 	this.speed = 100;
 16 	if ( this.domElement === document ) {
 17 		this.viewHalfX = window.innerWidth / 2;
 18 		this.viewHalfY = window.innerHeight / 2;
 19 	} else {
 20 		this.viewHalfX = this.domElement.offsetWidth / 2;
 21 		this.viewHalfY = this.domElement.offsetHeight / 2;
 22 		this.domElement.setAttribute( 'tabindex', -1 );}
 23 	this.domElement.addEventListener( 'mousemove', AICRAFT.bind( this, this.onMouseMove ), false );
 24 	this.domElement.addEventListener( 'mousedown', AICRAFT.bind( this, this.onMouseDown ), false );
 25 	this.domElement.addEventListener( 'mouseup', AICRAFT.bind( this, this.onMouseUp ), false );
 26 };

 30 AICRAFT.CameraControl.prototype.update = function() {
 31 	this.camera.position.x = this.gameObj.position.x;
 32 	this.camera.position.y = this.gameObj.position.y;
 33 	this.camera.position.z = this.gameObj.position.z;
 34 	var tailVector = this.tailVector();
 35 	this.camera.position.x += tailVector.x;
 36 	this.camera.position.y += tailVector.y;
 37 	this.camera.position.z += tailVector.z;
 38 	this.camera.position.y += 10;
 39 	this.camera.quaternion.x = this.gameObj.quaternion.x;
 40 	this.camera.quaternion.y = this.gameObj.quaternion.y;
 41 	this.camera.quaternion.z = this.gameObj.quaternion.z;
 42 	this.camera.quaternion.w = this.gameObj.quaternion.w;
 43 };

 45 AICRAFT.CameraControl.prototype.onMouseMove = function(event) {
 46 	if ( this.domElement === document ) {
 47 		this.mouseX = event.pageX - this.viewHalfX;
 48 		this.mouseY = event.pageY - this.viewHalfY;
 49 	} else {
 50 		this.mouseX = event.pageX - this.domElement.offsetLeft - this.viewHalfX;
 51 		this.mouseY = event.pageY - this.domElement.offsetTop - this.viewHalfY;
 52 	}
 53 	//flip Y axis
 54 	this.mouseY = this.mouseY*-1;
 55 	//convert to percentage
 56 	this.mouseX = this.mouseX / this.viewHalfX;
 57 	this.mouseY = this.mouseY / this.viewHalfY;
 58 	this.deltaX = this.mouseX - this.prevMouseX;
 59 	this.deltaY = this.mouseY - this.prevMouseY;
 60 
 61 	//insert code to mouse look
 62 	/*
 63 	if (this.mouseDragOn === true) {
 64 		this.gameObj.rotate(deltaX*this.speed, true);}
 65 		*/
 66 
 67 
 68 	this.prevMouseX = this.mouseX;
 69 	this.prevMouseY = this.mouseY;
 70 };

 72 AICRAFT.CameraControl.prototype.onMouseDown = function(event) {
 73 	if ( this.domElement !== document ) {
 74 		this.domElement.focus();
 75 	}
 76 	event.preventDefault();
 77 	event.stopPropagation();
 78 	this.mouseDragOn = true;
 79 };

 81 AICRAFT.CameraControl.prototype.onMouseUp = function(event) {

 84 		this.mouseDragOn = false;

 98 AICRAFT.CameraControl.setVector = function(self, distance, IsFront) {
 99 	if (IsFront === true) { var k = -1; } else { var k = 1;}
100 	var vector = new Ammo.btVector3(0,0,k);
101 	var quat = self.gameObj.phybody.getOrientation();
102 	var transform = new Ammo.btTransform();
103 	transform.setIdentity();
104 	transform.setRotation(quat);
105 	vector = transform.op_mul(vector);
106 	return new THREE.Vector3(vector.getX()*distance,
107 			vector.getY()*distance,
108 			vector.getZ()*distance);
109 };
