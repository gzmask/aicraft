1 AICRAFT.CodeEmitter = function(cameraControls, player, ai, socket, domElement) {
  2 	var self = this;
  3 	this.cameraControls = cameraControls;
  4 	this.socket = socket;
  5 	this.player = player;
  6 	this.ai = ai;
  7 	if (domElement === undefined || domElement === null) {
  8 		domElement = document.body;
  9 	}
 10 	this.IsEnable = false;
 11 	this.switching = false;
 12 	this.editor = document.createElement('div');
 13 	this.editor.id = "editor";
 14 	this.editor.style.background = '#999';
 15 	this.editor.style.width = '80%';
 16 	this.editor.style.height = '80%';
 17 	this.editor.style.zIndex = '-3';
 18 	this.editor.style.position = 'absolute';
 19 	this.editor.style.visibility = 'hidden';
 20 	this.editor.style.left =this.cameraControls.viewHalfX;
 21 	this.editor.style.top =this.cameraControls.viewHalfY;
 22 	//set up code emitter form
 23 	domElement.appendChild(this.editor);
 24 	this.editorAce = ace.edit('editor');
 25 	this.editorAce.setReadOnly(false);
 26 	this.editorAceDom = document.getElementById('editor');
 27 	socket.on('emitterInit', function(data){
 28 		data = data.replace(/ai_name_to_replace/g, 'AI_'+self.ai.name.toString());
 29 		self.editorAce.focus();
 30 		self.editorAce.getSession().setValue(data);
 31 	});
 32 };
 33 
 34 AICRAFT.CodeEmitter.prototype.constructor = AICRAFT.CodeEmitter;
 35 
 36 AICRAFT.CodeEmitter.prototype.request = function() {
 37 	if (this.switching === true) {return;}
 38 	this.switching = true;
 39 	if (this.IsEnable === true) {
 40 		this.send();
 41 	} else {
 42 		this.enable();
 43 	}
 44 	var self = this;
 45 	setTimeout(function(){
 46 		self.switching = false;
 47 	},1500);
 48 };
 49 
 50 AICRAFT.CodeEmitter.prototype.enable = function() {
 51 	if (this.IsEnable === true) {return;}
 52 	this.IsEnable = true;
 53 	this.editor.style.visibility = 'visible';
 54 	this.editor.style.left = '20%';
 55 	this.editor.style.top = '20%';
 56 	this.editorAceDom.style.visibility = 'visible';
 57 	this.editor.style.zIndex = '3';
 58     this.editor.focus();
 59 	this.editorAce.setReadOnly(false);
 60     this.editorAce.focus();
 61 	this.ai.codeUploading = true;
 62 	this.player.codeUploading = true;
 63 	this.socket.emit('code');
 64 };
 65 
 66 AICRAFT.CodeEmitter.prototype.send = function() {
 67 	if (this.IsEnable === false) {return;}
 68 	this.IsEnable = false;
 69 	this.editor.style.visibility = 'hidden';
 70 	this.editorAceDom.style.visibility = 'hidden';
 71 	this.editor.style.zIndex = '-3';
 72 	this.editorAce.setReadOnly(true);
 73 	this.ai.codeUploading = false;
 74 	this.player.codeUploading = false;
 75 	this.socket.emit('coded', this.editorAce.getSession().getValue());
 76 };
